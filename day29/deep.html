Complete SSRF URL validator
with DNS rebinding protection

ssrf-validator.ts — production-grade URL validationTYPESCRIPT
import dns     from 'dns/promises';
import ipaddr  from 'ipaddr.js';

const BLOCKED_RANGES = [
  'loopback',        // 127.0.0.0/8, ::1
  'private',         // 10.x, 172.16-31.x, 192.168.x
  'linkLocal',       // 169.254.0.0/16 (AWS metadata)
  'loopback',
  'multicast',
  'reserved',
  'unspecified',
];

export async function validateWebhookUrl(
  urlStr: string,
  options: { allowedDomains?: string[] } = {}
): Promise<{ safe: boolean; reason?: string }> {
  // 1. Parse URL
  let url: URL;
  try {
    url = new URL(urlStr);
  } catch {
    return { safe: false, reason: 'invalid_url' };
  }

  // 2. Protocol check — HTTPS only
  if (url.protocol !== 'https:') {
    return { safe: false, reason: 'https_required' };
  }

  // 3. Domain allowlist (if provided)
  if (options.allowedDomains?.length) {
    const allowed = options.allowedDomains.some(
      d => url.hostname === d || url.hostname.endsWith('.' + d)
    );
    if (!allowed) return { safe: false, reason: 'domain_not_allowed' };
  }

  // 4. Resolve hostname to IPs (catches DNS rebinding)
  let addresses: { address: string }[];
  try {
    addresses = await dns.lookup(url.hostname, { all: true });
  } catch {
    return { safe: false, reason: 'dns_resolution_failed' };
  }

  if (!addresses.length) {
    return { safe: false, reason: 'no_addresses' };
  }

  // 5. Check every resolved IP against blocked ranges
  for (const { address } of addresses) {
    try {
      const ip = ipaddr.parse(address);
      const range = ip.range();
      if (BLOCKED_RANGES.includes(range)) {
        return { safe: false, reason: `blocked_ip_range:${range}` };
      }
    } catch {
      return { safe: false, reason: 'ip_parse_failed' };
    }
  }

  return { safe: true };
}

// Usage with timeout and no redirect following
async function callWebhook(url: string, payload: object) {
  const { safe, reason } = await validateWebhookUrl(url);
  if (!safe) throw new Error(`Webhook URL blocked: ${reason}`);

  return fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
    signal: AbortSignal.timeout(5000),
    redirect: 'error',   // don't follow redirects (they may lead to private IPs)
  });
}

schema-first request validation

validation.ts — schemas for every input classTYPESCRIPT
import { z } from 'zod';

// UUID validation — prevents IDOR via guessable IDs
const UUIDSchema = z.string().uuid();

// Booking creation — strict schema, no unknown fields
export const CreateBookingSchema = z.object({
  eventId:  z.string().uuid(),
  seatId:   z.string().uuid(),
  quantity: z.number().int().min(1).max(10),
}).strict();   // .strict() rejects unknown fields (prevents mass assignment)

// Profile update — allowlist only
export const UpdateProfileSchema = z.object({
  name:        z.string().min(1).max(100).optional(),
  phoneNumber: z.string().regex(/^\+\d{10,15}$/).optional(),
  // role, isAdmin, planTier — NOT included (cannot be updated by user)
}).strict();

// Webhook registration — URL validated separately
export const WebhookSchema = z.object({
  url:        z.string().url().max(2048),
  eventTypes: z.array(z.enum(['booking.confirmed', 'booking.cancelled', 'payment.failed']))
               .min(1),
}).strict();

// Validation middleware
function validate<T>(schema: z.ZodSchema<T>) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({
        error: 'validation_failed',
        details: result.error.flatten().fieldErrors,
      });
    }
    req.body = result.data;  // replace with validated + typed data
    next();
  };
}

// Routes with validation
app.post('/bookings', authenticate, validate(CreateBookingSchema), createBooking);
app.put('/users/profile', authenticate, validate(UpdateProfileSchema), updateProfile);
app.post('/webhooks', authenticate, validate(WebhookSchema), registerWebhook);
