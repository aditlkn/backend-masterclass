Deep Dive 01

nginx upstream:
every parameter that matters

Most nginx upstream configs are 5 lines. A production config is 30 — the extra 25 lines prevent a specific category of failure at high traffic. Here's what each parameter does and what happens when it's wrong.

Complete production nginx upstream configurationNGINX
# /etc/nginx/conf.d/upstreams.conf

upstream api_backend {
    # Algorithm: least_conn uses P2C internally for multiple upstreams
    # Use least_conn for any workload with variable request duration
    least_conn;

    # Primary servers
    server 10.0.0.1:3000 weight=1 max_fails=3 fail_timeout=30s slow_start=30s;
    server 10.0.0.2:3000 weight=1 max_fails=3 fail_timeout=30s slow_start=30s;
    server 10.0.0.3:3000 weight=1 max_fails=3 fail_timeout=30s slow_start=30s;
    #
    # weight=1          — relative traffic share (all equal here)
    # max_fails=3       — passive: 3 errors within fail_timeout → mark down
    # fail_timeout=30s  — window for counting failures AND time server stays down
    # slow_start=30s    — ramp new/recovered server over 30s (nginx Plus only)

    # Backup server: only used when all primaries are down
    server 10.0.0.4:3000 backup;

    # keepalive: connection pool from nginx workers to upstream
    # This is NOT max_connections — it's idle connections kept warm per worker
    # Total connections = keepalive × worker_processes
    # Set to: (target_rps / worker_processes) × avg_response_time_s
    keepalive 32;

    # keepalive_requests: max requests per keepalive connection before close
    # Prevents memory leaks from long-lived upstream connections
    keepalive_requests 1000;

    # keepalive_time: max lifetime of a keepalive connection regardless of use
    keepalive_time 1h;

    # keepalive_timeout: close idle keepalive connections after this
    # Must be shorter than upstream server's keepalive timeout
    keepalive_timeout 60s;

    # queue: buffer requests when all upstreams are busy (nginx Plus)
    # Without this, nginx returns 502 immediately when all connections busy
    # queue 100 timeout=10s;  ← queue up to 100, wait up to 10s
}

server {
    listen 443 ssl http2;

    location /api/ {
        proxy_pass http://api_backend;

        # Timeout hierarchy (must be: nginx < client timeout)
        proxy_connect_timeout  3s;    # max time to establish TCP connection
        proxy_send_timeout     25s;   # max time between writes to upstream
        proxy_read_timeout     25s;   # max time waiting for upstream response
        # Client timeout (set in client) should be 30s
        # nginx times out at 25s → client gets 504 before its own 30s timeout
        # This gives client useful information rather than a silent drop

        # Required for keepalive to work
        proxy_http_version 1.1;
        proxy_set_header Connection "";  # clear "close" that HTTP/1.0 would add

        # Retry on specific errors (idempotent requests only)
        proxy_next_upstream error timeout http_502 http_503;
        proxy_next_upstream_tries 2;       # try 2 upstreams max
        proxy_next_upstream_timeout 5s;   # total retry budget
        # WARNING: never add http_500 here — retrying 500s can cause duplicates

        # Buffering: off for streaming, on for regular requests
        proxy_buffering on;
        proxy_buffer_size       4k;
        proxy_buffers           8 4k;
        proxy_busy_buffers_size 8k;
    }
}

# nginx stub_status — required for monitoring
server {
    listen 127.0.0.1:8080;
    location /nginx_status { stub_status; allow 127.0.0.1; deny all; }
}
💡
The keepalive parameter is per-worker, not global

keepalive 32 means 32 idle connections per nginx worker process, per upstream. If you have 4 worker processes and 3 upstream servers, you're keeping up to 4 × 3 × 32 = 384 connections warm. Make sure your upstream servers can handle this. For Node.js with a default connection limit of 1000, this is fine. For a service with a strict connection limit, calculate carefully.

Deep Dive 02

HAProxy runtime API:
zero-downtime server management

HAProxy's runtime API lets you modify the load balancer's behaviour without a reload — no config file change, no service restart, no dropped connections. This is how you do controlled deployments, gradual traffic shifts, and emergency server management in production.

Enabling the runtime API

The runtime API is a Unix socket that HAProxy exposes. You communicate with it via socat or any tool that can write to a socket. Enable it in the global section of your haproxy.cfg:

HAProxy runtime API — enable + complete command referenceBASH
# haproxy.cfg — enable the runtime API
# global:
#   stats socket /var/run/haproxy.sock mode 660 level admin expose-fd listeners
#   stats timeout 30s

## ── Helper function (add to your ~/.bashrc) ──────────────────────────
hactl() { echo "$@" | socat stdio /var/run/haproxy.sock; }

## ── View current state ───────────────────────────────────────────────
hactl "show servers state api_backend"
# Shows: server name, address, state (UP/DOWN/DRAIN), weight, sessions

hactl "show stat" | cut -d, -f1,2,18,19,48
# Columns: pxname, svname, status, weight, hrsp_5xx

## ── Traffic weight management ────────────────────────────────────────
# Shift traffic during canary deploy:
hactl "set weight api_backend/server-v2 10"   # 10% to new version
hactl "set weight api_backend/server-v2 50"   # 50% after validation
hactl "set weight api_backend/server-v2 100"  # full traffic
hactl "set weight api_backend/server-v1 0"    # drain v1

## ── Server state management ──────────────────────────────────────────
# Graceful drain: stop new connections, let existing complete
hactl "set server api_backend/server-1 state drain"
# Wait until sessions = 0, then:
hactl "set server api_backend/server-1 state maint"  # maintenance

# Emergency: immediately stop routing to a server
hactl "set server api_backend/server-1 state maint"

# Re-enable after fix:
hactl "set server api_backend/server-1 state ready"

## ── Slow-start workaround (HAProxy OSS — no native slow_start) ──────
# Add new server at weight 1, gradually raise via cron or script:
hactl "add server api_backend/server-4 10.0.0.4:3000"
hactl "set server api_backend/server-4 weight 5"
sleep 10
hactl "set server api_backend/server-4 weight 25"
sleep 10
hactl "set server api_backend/server-4 weight 100"

## ── Monitor sessions on a specific server ───────────────────────────
watch -n1 'echo "show servers state api_backend" | socat stdio /var/run/haproxy.sock'

## ── Clear error counters (after an incident) ────────────────────────
hactl "clear counters all"
Deep Dive 03

Health check endpoint design:
liveness vs readiness

Most teams have one health endpoint. You need two — and they should check different things. Conflating liveness and readiness is one of the most common causes of self-inflicted outages: a startup failing its dependency check fails the liveness probe, Kubernetes kills and restarts the pod in a loop, the dependency is actually fine but you've taken yourself down.

Liveness: is the process running?

The liveness probe answers: "should this process be killed and restarted?" It should only fail if the process is in a state where it cannot recover without a restart — infinite loop, deadlock, OOM. It should never check dependencies like databases or Redis. If the database is down and the liveness probe fails, Kubernetes kills all your pods simultaneously, compounding the database outage with a full application outage.

Liveness check: return 200 if the event loop is responsive. That's it.

Readiness: can this instance accept traffic?

The readiness probe answers: "should this pod receive traffic from the load balancer?" It fails during startup (dependencies not yet connected), during graceful shutdown (returning 503 tells the LB to drain), and when a required dependency is unavailable (database down — no point sending traffic to an instance that can't serve it).

Readiness check: verify all required dependencies. Return 503 with detail on what's failing.

The cascading failure trap

A database goes down. All pods fail their readiness check. The load balancer removes all pods from its pool. The database comes back up. But now all pods are simultaneously starting their readiness checks, all hitting the database at the same moment, causing a connection storm that overwhelms the recovering database. The readiness probe caused the cascade.

Fix: add a circuit breaker to your health check's dependency verification. If the dependency has been down for 30+ seconds, return 503 with "dependency_circuit_open": true. The LB removes you. When the circuit resets (dependency recovers), do a gradual re-add — check the dependency a few times with back-off before returning 200. This prevents the thundering herd on recovery.

Production health endpoints — liveness, readiness, and startup probesTYPESCRIPT
import { Router } from 'express';

const router = Router();
let isShuttingDown   = false;
let startupComplete  = false;
let dbCircuitOpen    = false;
let dbFailsSince: number | null = null;

// /ping — liveness only. Never checks dependencies.
// Kubernetes: livenessProbe. LB: TCP check on port.
router.get('/ping', (req, res) => {
  res.json({ status: 'alive', ts: Date.now() });
});

// /healthz — readiness. Checks dependencies. Used by LB and k8s readinessProbe.
router.get('/healthz', async (req, res) => {
  if (isShuttingDown)  return res.status(503).json({ status: 'shutting_down' });
  if (!startupComplete) return res.status(503).json({ status: 'starting' });

  const checks: Record<string, boolean | string> = {};
  let healthy = true;

  // Database — check with a short timeout
  try {
    await Promise.race([
      db.raw('SELECT 1'),
      new Promise((_, r) => setTimeout(() => r(new Error('timeout')), 2000)),
    ]);
    checks.database = true;
    dbFailsSince = null;
    dbCircuitOpen = false;
  } catch (err) {
    checks.database = false;
    healthy = false;
    if (!dbFailsSince) dbFailsSince = Date.now();
    // Open circuit after 30s of failures — prevents stampede on recovery
    if (Date.now() - (dbFailsSince ?? 0) > 30_000) {
      dbCircuitOpen = true;
      checks.database_circuit = 'open';
    }
  }

  // Redis — only if critical for your service
  try {
    await Promise.race([
      redis.ping(),
      new Promise((_, r) => setTimeout(() => r(new Error('timeout')), 1000)),
    ]);
    checks.redis = true;
  } catch {
    checks.redis = false;
    // Decide: is Redis critical? If session store, yes. If cache only, maybe not.
    // healthy = false;  ← uncomment if Redis is required for core functionality
  }

  return res.status(healthy ? 200 : 503).json({
    status: healthy ? 'ok' : 'degraded',
    checks,
    uptime: process.uptime(),
  });
});

// Mark startup complete after dependencies are warm
async function onAppReady() {
  await db.raw('SELECT 1'); // verify DB before accepting traffic
  startupComplete = true;
  logger.info('Application ready — accepting traffic');
}

// SIGTERM handler: fail readiness before closing
process.on('SIGTERM', async () => {
  isShuttingDown = true;    // /healthz → 503 immediately
  await sleep(5000);        // wait for LB to detect 503 and drain
  server.close();             // stop accepting new connections
  await db.destroy();        // close pool gracefully
  process.exit(0);
});
Verify your health endpoint design
✓/ping never checks dependencies — verify by disconnecting Redis/DB and confirming /ping still returns 200.
✓/healthz returns 503 when DB is disconnected — verify by temporarily blocking DB access.
✓Health check timeout is 2s or less — verify with time curl /healthz. Slow health checks throw off LB detection timing.
✓Response body includes which dependency failed — verify with curl /healthz | jq .checks. "degraded" with no detail is useless.
✓Graceful shutdown: /healthz returns 503 within 1s of SIGTERM — verify with kill -TERM $(pgrep node) && curl /healthz.
✓LB health check interval matches: if LB polls every 2s and your health check takes 2s, you have zero margin. Health check must complete in <50% of poll interval.
