Deep Dive 01

Pact consumer-driven contract testing

Consumers define what they expect. Producers verify they meet it. Breaking changes caught in CI before any service deploys.

email-service.pact.test.ts — consumer defines the contractTYPESCRIPT
import { PactV3, MatchersV3 } from '@pact-foundation/pact';
const { like, eachLike, string } = MatchersV3;

const provider = new PactV3({
  consumer: 'email-service',
  provider: 'booking-service',
  dir:      './pacts',
});

describe('email-service -> booking-service contract', () => {
  it('gets booking details for confirmation email', async () => {
    await provider
      .given('booking ord_123 exists and is confirmed')
      .uponReceiving('a request for booking details')
      .withRequest({
        method:  'GET',
        path:    '/v1/bookings/ord_123',
        headers: { Authorization: 'Bearer <token>' },
      })
      .willRespondWith({
        status: 200,
        body: {
          bookingId: string('ord_123'),
          status:    string('confirmed'),
          userId:    string('usr_456'),
          eventName: string('Coldplay Mumbai 2024'),
          venueName: string('DY Patil Stadium'),
          seatLabel: string('Block A, Row 3, Seat 12'),
          amountCents: like(49900),
          confirmedAt: string('2024-08-17T14:32:01Z'),
        },
      })
      .executeTest(async (mockServer) => {
        const client = new BookingClient(mockServer.url);
        const booking = await client.getBooking('ord_123');
        expect(booking.bookingId).toBe('ord_123');
        expect(booking.eventName).toBeDefined();
      });
  });
});

// booking-service/src/pact-verify.test.ts — producer verifies all contracts
import { Verifier } from '@pact-foundation/pact';

describe('booking-service contract verification', () => {
  it('satisfies all consumer contracts', () => {
    return new Verifier({
      provider:         'booking-service',
      providerBaseUrl:  'http://localhost:3000',
      pactBrokerUrl:    process.env.PACT_BROKER_URL,
      publishVerificationResult: true,
      providerVersion:  process.env.GIT_SHA,
      // State handlers — set up DB fixtures for each "given" clause
      stateHandlers: {
        'booking ord_123 exists and is confirmed': async () => {
          await db.query(`INSERT INTO bookings VALUES ('ord_123', 'confirmed', ...)`);
        },
      },
    }).verifyProvider();
    // Fails if booking-service does not return what email-service expects
  });
});
Deep Dive 02

Choreography saga:
booking flow without distributed transactions

booking-saga.ts — event-driven saga with compensating transactionsTYPESCRIPT
// Each service listens for events and publishes results
// No central orchestrator — pure choreography

// 1. checkout-service initiates
await kafka.publish('booking.initiated', {
  sagaId:    crypto.randomUUID(),  // trace the whole saga
  bookingId: 'ord_123',
  userId:    'usr_456',
  seatId:    'seat_789',
  amountCents: 49900,
});

// 2. inventory-service: handle seat hold
kafka.consume('booking.initiated', async (event) => {
  const held = await db.query(
    `UPDATE seats SET held_by=$1 WHERE id=$2 AND held_by IS NULL RETURNING id`,
    [event.bookingId, event.seatId]
  );
  if (held.rowCount === 0) {
    await kafka.publish('seat.hold.failed', { sagaId: event.sagaId, reason: 'seat_unavailable' });
    return;
  }
  await kafka.publish('seat.held', { sagaId: event.sagaId, seatId: event.seatId, bookingId: event.bookingId });
});

// 3. payment-service: handle charge
kafka.consume('seat.held', async (event) => {
  try {
    const charge = await stripe.charges.create({
      amount:   event.amountCents,
      currency: 'inr',
      metadata: { bookingId: event.bookingId, sagaId: event.sagaId },
    });
    await kafka.publish('payment.completed', { sagaId: event.sagaId, chargeId: charge.id });
  } catch (err) {
    // Payment failed — trigger compensating transaction
    await kafka.publish('payment.failed', { sagaId: event.sagaId, seatId: event.seatId });
  }
});

// 4a. booking-service: confirm on payment success
kafka.consume('payment.completed', async (event) => {
  await db.query(`UPDATE bookings SET status='confirmed' WHERE saga_id=$1`, [event.sagaId]);
  await kafka.publish('booking.confirmed', { sagaId: event.sagaId });
});

// 4b. inventory-service: release seat on payment failure (compensating tx)
kafka.consume('payment.failed', async (event) => {
  await db.query(`UPDATE seats SET held_by=NULL WHERE id=$1`, [event.seatId]);
  await kafka.publish('seat.released', { sagaId: event.sagaId });
});

// 4c. booking-service: cancel on seat unavailable
kafka.consume('seat.hold.failed', async (event) => {
  await db.query(`UPDATE bookings SET status='cancelled' WHERE saga_id=$1`, [event.sagaId]);
  await kafka.publish('booking.cancelled', { sagaId: event.sagaId, reason: 'seat_unavailable' });
});
