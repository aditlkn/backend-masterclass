Deep Dive 01

Pino production config:
serialisers, redaction, child loggers

Pino is the fastest Node.js JSON logger. It achieves speed by doing minimal work synchronously and offloading serialisation to a worker thread via pino-pretty or pino/transport. This config covers everything needed for production: request serialiser, error serialiser, sensitive field redaction, and the middleware pattern for child loggers.

logger.ts — complete Pino production configurationTYPESCRIPT
import pino, { Logger, SerializerFn } from 'pino';

// Custom serialisers — control exactly what gets logged for known types
const serializers: Record<string, SerializerFn> = {
  // err: pino's default is fine, but extend it
  err: pino.stdSerializers.err,

  // req: log the request, but NEVER the body or auth headers
  req: (req) => ({
    method:    req.method,
    path:      req.route?.path || req.path || req.url,  // route pattern not URL
    userAgent: req.headers?.['user-agent'],
    requestId: req.headers?.['x-request-id'],
    // deliberately omit: headers.authorization, body, query params
  }),

  // res: log response metadata, not body
  res: (res) => ({
    statusCode: res.statusCode,
    contentType: res.getHeader?.('content-type'),
    bytes: res.getHeader?.('content-length'),
  }),
};

// Fields to redact wherever they appear in the log object tree
const redactPaths = [
  'password', '*.password', 'body.password',
  'token', '*.token', 'accessToken', 'refreshToken',
  'cardNumber', '*.cardNumber', 'cvv',
  'authorization', '*.authorization', 'headers.authorization',
  'email',      // use userId instead
  'ssn', 'pan',  // national IDs
  'privateKey', 'secretKey', 'apiKey',
];

export const logger = pino({
  level: process.env.LOG_LEVEL || 'info',

  // Static metadata on every log line
  base: {
    service: process.env.SERVICE_NAME || 'unknown',
    version: process.env.APP_VERSION  || 'local',
    env:     process.env.NODE_ENV     || 'development',
    pid:     process.pid,
  },

  // ISO timestamp (not the default epoch ms)
  timestamp: pino.stdTimeFunctions.isoTime,

  // Emit "level" as string not number
  formatters: {
    level: (label) => ({ level: label }),
  },

  serializers,
  redact: { paths: redactPaths, censor: '[REDACTED]' },

  // In development: pretty print. In production: raw JSON to stdout.
  transport: process.env.NODE_ENV !== 'production'
    ? { target: 'pino-pretty', options: { colorize: true, translateTime: 'SYS:HH:MM:ss.l' } }
    : undefined,  // undefined = write JSON to stdout directly
});

// ── Express middleware: bind request context to child logger ─────
import { Request, Response, NextFunction } from 'express';
import { trace } from '@opentelemetry/api';

declare module 'express-serve-static-core' {
  interface Request { log: Logger; startTime: number; }
}

export function requestLogger() {
  return (req: Request, res: Response, next: NextFunction) => {
    req.startTime = Date.now();
    const span = trace.getActiveSpan();
    const traceCtx = span?.spanContext();

    req.log = logger.child({
      requestId: req.headers['x-request-id'] || crypto.randomUUID(),
      method:    req.method,
      path:      req.route?.path || req.path,
      ip:        req.ip,
      traceId:   traceCtx?.traceId,
      spanId:    traceCtx?.spanId,
    });

    // Suppress health check logs entirely
    if (req.path === '/healthz' || req.path === '/readyz') {
      req.log = { info: ()=>{}, warn: ()=>{}, error: ()=>{}, debug: ()=>{}, child: ()=>req.log } as any;
      return next();
    }

    res.on('finish', () => {
      const duration = Date.now() - req.startTime;
      const level = res.statusCode >= 500 ? 'error' : res.statusCode >= 400 ? 'warn' : 'info';

      // Sample: always log errors, sample 10% of successful requests
      if (level === 'info' && Math.random() > 0.10) return;

      req.log[level]({ statusCode: res.statusCode, durationMs: duration },
        'http.request.completed');
    });
    next();
  };
}

Grafana Loki + Promtail:
self-hosted log aggregation stack

Loki stores logs indexed only by labels (not content), keeping costs low. Promtail is the log shipper that reads Kubernetes pod logs from disk and adds labels from pod metadata. This Docker Compose stack runs the entire pipeline locally — identical to a Kubernetes production setup, just smaller.

Deep Dive 03

PII scrubbing and
field naming conventions

Two problems teams always hit late: PII in logs (a compliance and security issue discovered during an audit, not during development) and inconsistent field names across services (discovered during an incident when cross-service log queries return nothing because one service uses userId and another uses user_id). Fix both proactively.

sanitise.ts — strip PII before loggingTYPESCRIPT
// Use this before logging any object you don't fully control
// (request bodies, user-provided data, third-party API responses)

const SENSITIVE_KEYS = new Set([
  'password', 'passwd', 'secret', 'token', 'apiKey', 'api_key',
  'authorization', 'auth', 'credential', 'credentials',
  'cardNumber', 'card_number', 'cvv', 'cvc',
  'ssn', 'pan', 'aadhaar',
  'email', 'phone', 'mobile',
  'privateKey', 'private_key', 'secretKey',
]);

const SENSITIVE_PATTERNS = [
  /^\d{12,19}$/,           // card numbers (12-19 digits)
  /^\d{3,4}$/,             // CVV
  /^[A-Z0-9]{10,}$/,       // API keys (uppercase alphanum)
  /^Bearer\s+.+/i,         // auth tokens
  /^\d{12}$/,              // Aadhaar
];

export function sanitise(obj: unknown, depth = 0): unknown {
  if (depth > 10) return '[MAX_DEPTH]';  // prevent infinite recursion
  if (obj === null || typeof obj !== 'object') {
    if (typeof obj === 'string' && SENSITIVE_PATTERNS.some(p => p.test(obj))) {
      return '[REDACTED]';
    }
    return obj;
  }
  if (Array.isArray(obj)) {
    return obj.map(item => sanitise(item, depth + 1));
  }
  const result: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj as object)) {
    if (SENSITIVE_KEYS.has(key.toLowerCase())) {
      result[key] = '[REDACTED]';
    } else {
      result[key] = sanitise(value, depth + 1);
    }
  }
  return result;
}

// Usage: log user-provided data safely
logger.info({ body: sanitise(req.body) }, 'payment.request.received');
// Input:  { amount: 5000, cardNumber: "4111111111111111", cvv: "123" }
// Output: { amount: 5000, cardNumber: "[REDACTED]", cvv: "[REDACTED]" }

// ── Field naming conventions — document and enforce ───────────────
// Agree on these across ALL services before you have logs to query.

const LOG_FIELD_CONVENTIONS = {
  // IDs — always camelCase, always "Id" suffix
  userId:     'string',  // NOT user_id, user, userId, uid
  orderId:    'string',  // NOT order_id, orderID, order
  requestId:  'string',  // NOT request_id, reqId, req_id
  traceId:    'string',  // NOT trace_id, traceID

  // Time — always milliseconds as integer
  durationMs: 'number',  // NOT duration, duration_ms, "342ms" (string!)

  // Money — always lowest denomination as integer
  amountCents: 'number', // NOT amount, "₹50.00", amount_rupees
  currency:    'string', // ISO 4217: "INR", "USD", not "rupees"

  // HTTP
  statusCode:  'number', // NOT status, status_code, httpStatus
  method:      'string', // NOT httpMethod, verb
  path:        'string', // route pattern: /users/{id}, NOT /users/123

  // Events — dot.separated.hierarchy
  msg: 'payment.charge.failed',  // NOT "Payment charge failed", not camelCase
} as const;
