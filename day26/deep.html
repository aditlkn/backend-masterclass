-- Core events table
CREATE TABLE events (
  global_position  BIGSERIAL     PRIMARY KEY,
  aggregate_id     TEXT          NOT NULL,
  aggregate_type   TEXT          NOT NULL,
  version          INTEGER       NOT NULL,
  event_type       TEXT          NOT NULL,
  event_version    INTEGER       NOT NULL DEFAULT 1,  -- for upcasting
  payload          JSONB         NOT NULL,
  metadata         JSONB         NOT NULL DEFAULT '{}',
  created_at       TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- Optimistic concurrency: reject concurrent appends to same aggregate
ALTER TABLE events
  ADD CONSTRAINT events_aggregate_version_unique
  UNIQUE (aggregate_id, version);

-- Efficient aggregate loading
CREATE INDEX idx_events_aggregate
  ON events (aggregate_id, version ASC);

-- Projection catch-up: load events from a global position
CREATE INDEX idx_events_global_position
  ON events (global_position ASC);

-- Snapshots table
CREATE TABLE snapshots (
  aggregate_id  TEXT         PRIMARY KEY,
  version       INTEGER      NOT NULL,
  state         JSONB        NOT NULL,
  created_at    TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- Projection checkpoints
CREATE TABLE projection_checkpoints (
  name      TEXT    PRIMARY KEY,
  position  BIGINT  NOT NULL DEFAULT 0
);
event-store.ts — append with optimistic locking and loadTYPESCRIPT
export async function appendEvents(
  aggregateId:   string,
  aggregateType: string,
  events:        DomainEvent[],
  expectedVersion: number   // number of events already seen
): Promise<void> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    for (let i = 0; i < events.length; i++) {
      const version = expectedVersion + i + 1;
      try {
        await client.query(
          `INSERT INTO events
           (aggregate_id, aggregate_type, version, event_type, payload, metadata)
           VALUES ($1, $2, $3, $4, $5, $6)`,
          [aggregateId, aggregateType, version,
           events[i].type, events[i].payload, events[i].metadata ?? {}]
        );
      } catch (err: any) {
        if (err.code === '23505') {  // unique_violation
          await client.query('ROLLBACK');
          throw new ConcurrencyError(`Concurrent write on ${aggregateId} v${version}`);
        }
        throw err;
      }
    }
    await client.query('COMMIT');
  } finally {
    client.release();
  }
}

export async function loadAggregate<T>(
  aggregateId: string,
  apply:       (state: T, event: DomainEvent) => T,
  initial:     T
): Promise<{ state: T; version: number }> {
  // Try snapshot first
  const snap = await pool.query(
    'SELECT version, state FROM snapshots WHERE aggregate_id = $1',
    [aggregateId]
  );
  const fromVersion = snap.rows[0]?.version ?? 0;
  const baseState   = snap.rows[0]?.state   ?? initial;

  // Load events after snapshot
  const result = await pool.query(
    `SELECT version, event_type, event_version, payload
     FROM events
     WHERE aggregate_id = $1 AND version > $2
     ORDER BY version ASC`,
    [aggregateId, fromVersion]
  );

  const events = result.rows.map(row => upcast(row));
  const state  = events.reduce(apply, baseState);
  const version = result.rows.at(-1)?.version ?? fromVersion;

  return { state, version };
}

projection-runner.ts — catch-up then live subscriptionTYPESCRIPT
export async function runProjection(
  name:    string,
  handler: (event: DomainEvent) => Promise<void>,
  batchSize = 500
) {
  let position = await loadCheckpoint(name);
  logger.info({ name, position }, 'projection.starting');

  // Phase 1: catch-up from checkpoint to current head
  while (true) {
    const events = await pool.query(
      `SELECT global_position, aggregate_id, event_type, event_version, payload
       FROM events
       WHERE global_position > $1
       ORDER BY global_position ASC
       LIMIT $2`,
      [position, batchSize]
    );
    if (events.rows.length === 0) break;  // caught up

    for (const row of events.rows) {
      const event = upcast(row);
      try {
        await handler(event);
      } catch (err) {
        logger.error({ err, event }, 'projection.handler.failed');
        throw err;  // stop — do not advance checkpoint past failed event
      }
      position = row.global_position;
      await saveCheckpoint(name, position);
    }
    logger.info({ name, position }, 'projection.catchup.progress');
  }
  logger.info({ name, position }, 'projection.caught_up');

  // Phase 2: live — poll for new events
  while (true) {
    await sleep(100);  // or use LISTEN/NOTIFY for lower latency
    const events = await pool.query(
      `SELECT global_position, aggregate_id, event_type, event_version, payload
       FROM events WHERE global_position > $1
       ORDER BY global_position ASC LIMIT $2`,
      [position, batchSize]
    );
    for (const row of events.rows) {
      await handler(upcast(row));
      position = row.global_position;
      await saveCheckpoint(name, position);
    }
  }
}

// Upcaster — transform old event shapes to current shape on read
const upcasters: Record<string, Record<number, (p: any) => any>> = {
  'BookingConfirmed': {
    1: (p) => ({ ...p, confirmedAt: p.timestamp ?? p.confirmedAt }),
    // v1 used 'timestamp', v2 uses 'confirmedAt' — normalise on read
  },
};

function upcast(row: any): DomainEvent {
  let payload = row.payload;
  const chain = upcasters[row.event_type];
  if (chain) {
    for (let v = row.event_version; v < CURRENT_VERSION; v++) {
      if (chain[v]) payload = chain[v](payload);
    }
  }
  return { type: row.event_type, payload, aggregateId: row.aggregate_id };
}
