Deep Dive 01

DataLoader: batch and cache per-request

DataLoader batches all individual lookups within a single event loop tick into one database query, then distributes results back to each caller. The same pattern used by GraphQL's official DataLoader, usable in any context.

dataloader.ts — generic batching loaderTYPESCRIPT
export class DataLoader<K, V> {
  private queue: { key: K; resolve: (v: V) => void; reject: (e: Error) => void }[] = [];
  private cache = new Map<K, V>();
  private scheduled = false;

  constructor(
    private batchFn: (keys: K[]) => Promise<Map<K, V>>
  ) {}

  async load(key: K): Promise<V> {
    if (this.cache.has(key)) return this.cache.get(key)!;

    return new Promise((resolve, reject) => {
      this.queue.push({ key, resolve, reject });
      if (!this.scheduled) {
        this.scheduled = true;
        // Defer execution to next tick — collects all load() calls
        process.nextTick(() => this.dispatch());
      }
    });
  }

  private async dispatch() {
    const batch = this.queue.splice(0);
    this.scheduled = false;
    const keys = [...new Set(batch.map(b => b.key))];

    try {
      const results = await this.batchFn(keys);
      results.forEach((v, k) => this.cache.set(k, v));
      batch.forEach(({ key, resolve, reject }) => {
        const v = results.get(key);
        v !== undefined ? resolve(v) : reject(new Error(`Key not found: ${key}`));
      });
    } catch (err) {
      batch.forEach(({ reject }) => reject(err as Error));
    }
  }
}

// Create per-request (fresh cache per request)
function createLoaders() {
  return {
    venue: new DataLoader<string, Venue>(async (ids) => {
      const rows = await db.query(
        'SELECT * FROM venues WHERE id = ANY($1)', [ids]
      );
      return new Map(rows.rows.map(v => [v.id, v]));
    }),
    organiser: new DataLoader<string, Organiser>(async (ids) => {
      const rows = await db.query(
        'SELECT * FROM organisers WHERE id = ANY($1)', [ids]
      );
      return new Map(rows.rows.map(o => [o.id, o]));
    }),
  };
}

// Usage — even with per-event loads, only 2 queries total
app.get('/events', async (req, res) => {
  const loaders = createLoaders();                    // fresh per request
  const events = await db.query('SELECT * FROM events LIMIT 50');
  const enriched = await Promise.all(
    events.rows.map(async e => ({
      ...e,
      venue:     await loaders.venue.load(e.venueId),      // batched
      organiser: await loaders.organiser.load(e.organiserId), // batched
    }))
  );
  res.json(enriched);
  // Total queries: 1 (events) + 1 (all venues) + 1 (all organisers) = 3
});
Deep Dive 02

Reading EXPLAIN ANALYZE:
find the seq scan

EXPLAIN ANALYZE — what each line meansSQL
-- Run on a replica, not primary. Always add LIMIT.
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT e.*, o.name, v.name
FROM events e
JOIN organisers o ON o.id = e.organiser_id
JOIN venues    v ON v.id  = e.venue_id
WHERE e.starts_at > NOW()
ORDER BY e.starts_at
LIMIT 50;

-- What to look for:
-- "Seq Scan on events" with Filter removing 45,000 rows -> needs index on starts_at
-- "Seq Scan on venues" with 800 rows -> small table, seq scan is fine
-- "actual time=12.3ms" total -> acceptable for this query

-- Add the missing index:
CREATE INDEX CONCURRENTLY idx_events_starts_at ON events (starts_at);

-- CONCURRENTLY: builds the index without locking the table (takes longer but safe in prod)
