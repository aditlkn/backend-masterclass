The three implementation patterns that make JWT auth production-grade: automatic key rotation via JWKS, refresh token issuance and rotation, and the PKCE authorization code flow for web clients.

Deep Dive 01

JWKS key rotation:
zero-downtime public key updates

Instead of configuring each service with a static public key, services fetch the current public key from the auth server JWKS endpoint. Key rotation is publishing a new key alongside the old one, waiting for old tokens to expire, then removing the old key.

auth-middleware.ts — JWKS client with key cachingTYPESCRIPT
import jwksClient from 'jwks-rsa';
import jwt        from 'jsonwebtoken';

const client = jwksClient({
  jwksUri:  'https://auth.example.com/.well-known/jwks.json',
  cache:    true,
  cacheMaxEntries: 5,
  cacheMaxAge:     600_000,   // 10 minutes — re-fetch keys periodically
  rateLimit: true,
  jwksRequestsPerMinute: 10, // avoid hammering auth server
});

function getKey(header, callback) {
  client.getSigningKey(header.kid, (err, key) => {
    callback(err, key?.getPublicKey());
  });
}

export function authenticate(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'missing_token' });

  jwt.verify(token, getKey, {
    algorithms: ['RS256'],         // explicit — never omit
    audience:   'booking-api',
    issuer:     'https://auth.example.com',
  }, (err, decoded) => {
    if (err) return res.status(401).json({ error: err.message });
    req.user = decoded;
    next();
  });
}

// Key rotation procedure (auth server side):
// 1. Generate new key pair: openssl genrsa -out private.pem 2048
// 2. Add new public key to JWKS response (alongside old key)
// 3. Start signing new tokens with new private key
// 4. Wait for all old tokens to expire (max_expiry = 15 min)
// 5. Remove old public key from JWKS response
// Services automatically pick up new key via JWKS cache refresh
Deep Dive 02

Refresh token implementation:
rotation and secure storage

auth-server.ts — refresh token issuance and rotationTYPESCRIPT
import jwt    from 'jsonwebtoken';
import crypto from 'crypto';

// Issue tokens on login
app.post('/auth/login', async (req, res) => {
  const user = await db.verifyCredentials(req.body);
  if (!user) return res.status(401).json({ error: 'invalid_credentials' });

  const accessToken = jwt.sign(
    { sub: user.id, roles: user.roles, jti: crypto.randomUUID() },
    privateKey,
    { algorithm: 'RS256', expiresIn: '15m',
      audience: 'booking-api', issuer: 'https://auth.example.com' }
  );

  const refreshTokenValue = crypto.randomBytes(32).toString('hex');
  await db.createRefreshToken({
    token:     crypto.createHash('sha256').update(refreshTokenValue).digest('hex'),
    userId:    user.id,
    expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000),
  });

  res
    .cookie('refresh_token', refreshTokenValue, {
      httpOnly: true,   // inaccessible to JS
      secure:   true,   // HTTPS only
      sameSite: 'strict',
      maxAge:   7 * 24 * 60 * 60 * 1000,
      path:     '/auth', // only sent to /auth endpoints
    })
    .json({ access_token: accessToken });
});

// Rotate refresh token on use
app.post('/auth/refresh', async (req, res) => {
  const rawToken = req.cookies.refresh_token;
  if (!rawToken) return res.status(401).json({ error: 'no_refresh_token' });

  const tokenHash = crypto.createHash('sha256').update(rawToken).digest('hex');
  const stored = await db.findRefreshToken(tokenHash);

  if (!stored || stored.revoked || stored.expiresAt < new Date()) {
    // Possible token reuse attack — revoke all tokens for this user
    if (stored?.revoked) await db.revokeAllUserTokens(stored.userId);
    return res.status(401).json({ error: 'invalid_refresh_token' });
  }

  // Rotate: revoke old, issue new
  await db.revokeRefreshToken(tokenHash);

  const user = await db.findUser(stored.userId);

  // ... (same code as login)
});
