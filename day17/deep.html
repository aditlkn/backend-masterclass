Deep Dive 01

Production RateLimiter class:
both algorithms, one interface

A rate limiter scattered across the codebase — different Lua scripts, different key formats, different header logic — is impossible to audit or change consistently. This class wraps both token bucket and sliding window behind a single interface, centralises all Lua scripts, and handles headers and error responses uniformly.

RateLimiter class — token bucket + sliding window, typedTYPESCRIPT
import { Redis } from 'ioredis';
import { Request, Response, NextFunction } from 'express';

interface RLResult {
  allowed: boolean;
  remaining: number;
  limit: number;
  resetAt: number;    // Unix timestamp
  retryAfter?: number; // seconds until retry
}

export class RateLimiter {
  private tokenBucketScript: string;
  private slidingWindowScript: string;

  constructor(private redis: Redis) {
    // Load Lua scripts once (SHA cached by Redis)
    this.tokenBucketScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local rate = tonumber(ARGV[2])
      local requested = tonumber(ARGV[3])
      local now = tonumber(redis.call('TIME')[1])
      local b = redis.call('HMGET', key, 'tokens', 'ts')
      local tokens = tonumber(b[1]) or capacity
      local ts = tonumber(b[2]) or now
      local elapsed = math.max(0, now - ts)
      tokens = math.min(capacity, tokens + elapsed * rate)
      local reset_in = math.ceil((capacity - tokens) / rate)
      if tokens < requested then
        redis.call('HMSET', key, 'tokens', tokens, 'ts', now)
        redis.call('EXPIRE', key, math.ceil(capacity / rate) + 10)
        return {0, math.floor(tokens), capacity, now + reset_in}
      end
      tokens = tokens - requested
      redis.call('HMSET', key, 'tokens', tokens, 'ts', now)
      redis.call('EXPIRE', key, math.ceil(capacity / rate) + 10)
      return {1, math.floor(tokens), capacity, now + reset_in}
    `;

    this.slidingWindowScript = `
      local curr_key = KEYS[1]
      local prev_key = KEYS[2]
      local window = tonumber(ARGV[1])
      local limit = tonumber(ARGV[2])
      local now = tonumber(redis.call('TIME')[1])
      local curr = tonumber(redis.call('GET', curr_key)) or 0
      local prev = tonumber(redis.call('GET', prev_key)) or 0
      local window_start = math.floor(now / window) * window
      local elapsed_frac = (now - window_start) / window
      local count = prev * (1 - elapsed_frac) + curr
      local reset_at = window_start + window
      if count >= limit then
        return {0, 0, limit, reset_at}
      end
      redis.call('INCR', curr_key)
      redis.call('EXPIRE', curr_key, window * 2)
      return {1, math.floor(limit - count - 1), limit, reset_at}
    `;
  }

  async tokenBucket(key: string, capacity: number, rate: number, cost = 1): Promise<RLResult> {
    const result = await this.redis.eval(
      this.tokenBucketScript, 1,
      `rl:tb:\${key}`, capacity, rate, cost
    ) as [number, number, number, number];

    return {
      allowed:    result[0] === 1,
      remaining:  result[1],
      limit:      result[2],
      resetAt:    result[3],
      retryAfter: result[0] === 0 ? result[3] - Math.floor(Date.now() / 1000) : undefined,
    };
  }

  async slidingWindow(key: string, limit: number, windowSec: number): Promise<RLResult> {
    const now = Math.floor(Date.now() / 1000);
    const windowKey = Math.floor(now / windowSec);
    const result = await this.redis.eval(
      this.slidingWindowScript, 2,
      `rl:sw:\${key}:\${windowKey}`,
      `rl:sw:\${key}:\${windowKey - 1}`,
      windowSec, limit
    ) as [number, number, number, number];

    return {
      allowed:    result[0] === 1,
      remaining:  result[1],
      limit:      result[2],
      resetAt:    result[3],
      retryAfter: result[0] === 0 ? result[3] - now : undefined,
    };
  }

  // Express middleware factory
  middleware(
    keyFn: (req: Request) => string,
    opts: { algorithm: 'token_bucket' | 'sliding_window'; limit: number; window?: number; rate?: number; cost?: number }
  ) {
    return async (req: Request, res: Response, next: NextFunction) => {
      const key = keyFn(req);
      const result = opts.algorithm === 'token_bucket'
        ? await this.tokenBucket(key, opts.limit, opts.rate!, opts.cost)
        : await this.slidingWindow(key, opts.limit, opts.window!);

      res.set({
        'X-RateLimit-Limit':     opts.limit,
        'X-RateLimit-Remaining': result.remaining,
        'X-RateLimit-Reset':     result.resetAt,
      });

      if (!result.allowed) {
        res.set('Retry-After', String(result.retryAfter));
        return res.status(429).json({
          error: 'rate_limit_exceeded',
          limit: opts.limit,
          retry_after: result.retryAfter,
          docs: 'https://api.example.com/docs/rate-limits',
        });
      }
      next();
    };
  }
}
Verify rate limiting is working
✓429 responses include Retry-After: curl -v POST /api/auth/login repeatedly — 429 response must have Retry-After: N header.
✓Lua atomicity: run 100 concurrent requests at the limit boundary — the count should never exceed the limit by more than 0.
✓Redis key TTLs are set: redis-cli TTL "rl:tb:user:123" — should return a positive number, never -1 (no TTL = memory leak).
✓Metrics track 429s: check your monitoring dashboard for a rate_limit_exceeded metric. Alert threshold set at 10× baseline per endpoint.
Deep Dive 02

Kong rate-limiting:
declarative config with per-consumer overrides

Kong's rate-limiting plugin implements the sliding window counter algorithm in Lua. The declarative config approach lets you define different limits per consumer (free vs paid), set multiple time windows (per-minute AND per-hour), and use Redis for distributed state — all without application code changes.

Kong rate limiting — complete declarative config (kong.yaml)YAML
# kong.yaml — rate limiting at the gateway layer
_format_version: "3.0"

## ── Global rate limit (applies to all routes without override) ───
plugins:
  - name: rate-limiting
    config:
      minute: 60           # 60 requests per minute (global default)
      hour:   1000         # 1000 per hour
      policy: redis
      redis_host: redis.prod.internal
      redis_port: 6379
      limit_by: consumer   # limit per authenticated consumer
      fault_tolerant: true # allow requests if Redis is unavailable
      hide_client_headers: false  # send X-RateLimit-* headers

## ── Services and routes ──────────────────────────────────────────
services:
  - name: api-service
    host: api.prod.internal
    port: 8080
    routes:
      - name: api-routes
        paths: [/api/v1]
        plugins:
          # Stricter limit on expensive endpoints
          - name: rate-limiting
            config:
              minute: 30
              hour: 500
              policy: redis
              redis_host: redis.prod.internal
              limit_by: consumer

      - name: auth-routes
        paths: [/api/v1/auth]
        plugins:
          # Very strict on auth — 5 per 15 minutes
          - name: rate-limiting
            config:
              minute: 5       # 5 per minute per IP
              limit_by: ip    # IP-based for unauthenticated auth endpoint
              policy: redis
              redis_host: redis.prod.internal
              fault_tolerant: false  # fail-closed for auth — security critical

## ── Consumers with plan-based overrides ─────────────────────────
consumers:
  - username: free-tier
    plugins:
      - name: rate-limiting
        config:
          minute: 10
          hour: 100
          day: 1000
          policy: redis
          redis_host: redis.prod.internal

  - username: pro-tier
    plugins:
      - name: rate-limiting
        config:
          minute: 100
          hour: 5000
          day: 50000
          policy: redis
          redis_host: redis.prod.internal

  - username: enterprise-tier
    plugins:
      - name: rate-limiting
        config:
          minute: 1000
          hour: 50000
          day: 500000
          policy: redis
          redis_host: redis.prod.internal

  - username: internal-service
    plugins:
      - name: rate-limiting
        config:
          minute: 10000    # very high for internal services
          policy: redis
          redis_host: redis.prod.internal

## ── Apply changes ────────────────────────────────────────────────

Deep Dive 03

Tiered limits by subscription plan:
one middleware, all plans

Most SaaS APIs need different limits for different subscription tiers. Rather than writing separate middleware for each plan, a single plan-aware middleware reads the user's subscription from the request context and applies the appropriate limits — both the per-minute cap and the burst tolerance change with the plan.

Plan-aware rate limiter — one middleware, all tiersTYPESCRIPT
// Plan definitions — single source of truth
const PLAN_LIMITS = {
  free: {
    requestsPerHour: 100,
    burstCapacity:   10,    // token bucket capacity
    refillPerMin:    1.67,  // 100/hr = 1.67/min
    costMultiplier:  1,     // all endpoints cost 1 token
  },
  pro: {
    requestsPerHour: 5000,
    burstCapacity:   200,
    refillPerMin:    83.3,
    costMultiplier:  1,
  },
  enterprise: {
    requestsPerHour: 50000,
    burstCapacity:   2000,
    refillPerMin:    833,
    costMultiplier:  1,
  },
} as const;

// Endpoint costs — expensive endpoints cost more tokens
const ENDPOINT_COSTS: Record<string, number> = {
  '/api/v1/ai/generate':   10,  // LLM endpoint: costs 10 tokens
  '/api/v1/reports/export': 5,   // heavy export: 5 tokens
  '/api/v1/bulk':           3,   // bulk operations: 3 tokens
  'default':                1,   // all other endpoints: 1 token
};

function planRateLimiter(rl: RateLimiter) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const user = req.user;  // set by auth middleware
    if (!user) return next();  // unauthenticated: handled by IP limit at gateway

    const plan = PLAN_LIMITS[user.subscription as keyof typeof PLAN_LIMITS]
      || PLAN_LIMITS.free;

    // Endpoint cost (expensive operations cost more tokens)
    const path = req.route?.path || req.path;
    const cost = ENDPOINT_COSTS[path] || ENDPOINT_COSTS['default'];
    const effectiveCost = cost * plan.costMultiplier;

    const result = await rl.tokenBucket(
      `user:\${user.id}`,
      plan.burstCapacity,
      plan.refillPerMin / 60,  // convert to per-second
      effectiveCost
    );

    res.set({
      'X-RateLimit-Limit':     plan.burstCapacity,
      'X-RateLimit-Remaining': result.remaining,
      'X-RateLimit-Reset':     result.resetAt,
      'X-RateLimit-Plan':      user.subscription,  // helps debugging
    });

    if (!result.allowed) {
      // Include upgrade suggestion in the error response
      const nextPlan = user.subscription === 'free' ? 'pro'
                     : user.subscription === 'pro'  ? 'enterprise'
                     : null;
      res.set('Retry-After', String(result.retryAfter));
      return res.status(429).json({
        error:       'rate_limit_exceeded',
        plan:        user.subscription,
        retry_after: result.retryAfter,
        ...(nextPlan && {
          upgrade_message: `Upgrade to \${nextPlan} for \${PLAN_LIMITS[nextPlan as keyof typeof PLAN_LIMITS].requestsPerHour} req/hr`,
          upgrade_url: `https://app.example.com/billing/upgrade?to=\${nextPlan}`,
        }),
      });
    }
    next();
  };
}

// Usage
const limiter = new RateLimiter(redis);
app.use('/api', authMiddleware, planRateLimiter(limiter));

// Auth endpoint gets its own strict limiter (before plan check)
app.use('/api/auth/login',
  limiter.middleware(req => req.ip, {
    algorithm: 'sliding_window', limit: 5, window: 900
  })
);
Verify tiered limits work correctly
✓Free user hits limit at 10 burst: make 11 rapid requests as free user — 11th should return 429 with upgrade_message.
✓Expensive endpoints cost more: call /api/v1/ai/generate as free user — should deplete 10 tokens per call, hitting limit after 1 request.
✓Pro user has 200-token burst: pro user should handle 200 rapid requests before hitting limit.
✓429 body includes retry_after: curl -s -o /dev/null -D - POST /api/v1/ai/generate when rate limited — confirm Retry-After header and upgrade_url in body.
