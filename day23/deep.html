KafkaJS: complete producer and consumer

The full setup — idempotent producer with acks=all, manual-commit consumer with deduplication, graceful shutdown, and structured logging on every event.

kafka.ts — producer with acks=all and idempotencyTYPESCRIPT
import { Kafka, CompressionTypes, logLevel } from 'kafkajs';
import { logger } from './logger';
import { redis }  from './redis';

export const kafka = new Kafka({
  clientId: process.env.SERVICE_NAME || 'checkout-service',
  brokers:  (process.env.KAFKA_BROKERS || 'kafka:9092').split(','),
  ssl: process.env.NODE_ENV === 'production',
  logLevel: logLevel.WARN,
  retry: { initialRetryTime: 300, retries: 8 },
});

// ── Producer ─────────────────────────────────────────────
const producer = kafka.producer({
  allowAutoTopicCreation: false,
  idempotent:            true,   // exactly-once producer
  transactionalId:       `\${process.env.SERVICE_NAME}-producer`,
});

export async function publishEvent(
  topic:   string,
  key:     string,
  payload: unknown,
  headers: Record<string, string> = {}
): Promise<void> {
  await producer.send({
    topic,
    compression: CompressionTypes.GZIP,
    messages: [{
      key,
      value:   JSON.stringify(payload),
      headers: {
        'schema-version': '1',
        'source-service': process.env.SERVICE_NAME || 'unknown',
        ...headers,
      },
    }],
  });
  logger.info({ topic, key, payloadSize: JSON.stringify(payload).length },
    'kafka.event.published');
}

// ── Consumer ─────────────────────────────────────────────
export function createConsumer(groupId: string) {
  return kafka.consumer({
    groupId,
    sessionTimeout:        30000,
    heartbeatInterval:     3000,
    maxPollIntervalMs:     300000,
    allowAutoTopicCreation:false,
  });
}

export async function runConsumer(
  groupId: string,
  topics:  string[],
  handler: (topic: string, key: string, value: unknown) => Promise<void>
) {
  const consumer = createConsumer(groupId);
  await consumer.connect();
  await consumer.subscribe({ topics, fromBeginning: false });

  await consumer.run({
    autoCommit: false,   // manual commit — always
    eachMessage: async ({ topic, partition, message, heartbeat }) => {
      const key   = message.key?.toString()   || '';
      const value = message.value?.toString() || '{}';
      const eventId = `\${topic}:\${partition}:\${message.offset}`;

      // Idempotency check — skip if already processed
      const claimed = await redis.set(
        `kafka:processed:\${eventId}`, '1', 'EX', 86400, 'NX'
      );
      if (!claimed) {
        logger.info({ eventId }, 'kafka.event.duplicate.skipped');
        return;
      }

      try {
        const parsed = JSON.parse(value);
        await heartbeat();  // prevent session timeout on slow processing
        await handler(topic, key, parsed);
        logger.info({ topic, partition, offset: message.offset, key },
          'kafka.event.processed');
      } catch (err) {
        await redis.del(`kafka:processed:\${eventId}`);  // release lock on failure
        logger.error({ err, topic, partition, offset: message.offset, key },
          'kafka.event.processing.failed');
        throw err;  // rethrow — consumer will retry or stall
      }
    },
  });

  // Graceful shutdown
  const shutdown = async () => {
    logger.info('Disconnecting Kafka consumer...');
    await consumer.disconnect();
    process.exit(0);
  };
  process.on('SIGTERM', shutdown);
  process.on('SIGINT',  shutdown);
}
Verify
✓Auto-commit disabled: check consumer config includes autoCommit: false. Verify by killing the consumer mid-batch and confirming events are redelivered on restart.
✓Idempotency working: send the same event twice (same topic/partition/offset key). Second processing attempt should log kafka.event.duplicate.skipped.
✓acks=all verified: check producer config has idempotent: true (which implies acks=all). Confirm with kafka-configs.sh --describe --topic booking.confirmed — min.insync.replicas should be 2.

Consumer lag monitoring:
Prometheus + Grafana alert

lag-monitor.ts — expose consumer lag as Prometheus metricsTYPESCRIPT
import { kafka } from './kafka';
import { Gauge } from 'prom-client';

const lagGauge = new Gauge({
  name:       'kafka_consumer_lag',
  help:       'Consumer group lag per partition',
  labelNames: ['group', 'topic', 'partition'],
});

async function collectLag() {
  const admin = kafka.admin();
  await admin.connect();

  const groups = ['email-service', 'analytics-service', 'inventory-service'];

  for (const groupId of groups) {
    const offsets = await admin.fetchOffsets({ groupId, topics: ['booking.confirmed'] });
    const topicOffsets = await admin.fetchTopicOffsets('booking.confirmed');

    for (const { partition, offset: committed } of offsets[0].partitions) {
      const latest = topicOffsets.find(o => o.partition === partition)?.offset || '0';
      const lag = parseInt(latest) - parseInt(committed);
      lagGauge.set({ group: groupId, topic: 'booking.confirmed', partition: String(partition) }, lag);
    }
  }
  await admin.disconnect();
}

setInterval(collectLag, 15_000);  // collect every 15s
